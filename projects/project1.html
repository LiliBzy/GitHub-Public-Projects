import random output = Element("output") def print_msg(message): output.write(message + "\n") def guessing_game(): num = random.randint(1, 10) tries = 3 print_msg("🎯 I'm thinking of a number between 1 and 10. You have 3 attempts.") while tries > 0: guess = int(prompt("Enter your guess (1–10):")) if not 1 <= guess <= 10: print_msg("🚫 Out of range. Try again.") continue if guess == num: print_msg("🎉 You win!") break tries -= 1 hint = "too high" if guess > num else "too low" print_msg(f"❌ Your guess is {hint}. {tries} attempts left.") else: print_msg(f"💀 Game Over. The correct number was {num}.") replay = prompt("🔁 Play again? (Y/N):").strip().lower() if replay in ("y", "yes"): print_msg("Restarting...\n") guessing_game() else: print_msg("👋 Thanks for playing!") guessing_game()
