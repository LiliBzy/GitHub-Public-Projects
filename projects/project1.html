🎯 Guess the Number

I'm thinking of a number between 1 and 10. You have 3 attempts.

from js import document import random number = random.randint(1, 10) tries = 3 def update_output(message): document.getElementById("output").innerText = message def handle_guess(event=None): global tries, number try: guess = int(document.getElementById("guess").value) except: update_output("🚫 Please enter a valid number.") return if guess < 1 or guess > 10: update_output("🚫 Number out of range. Try again.") return if guess == number: update_output("🎉 You win! The number was " + str(number)) tries = 0 else: tries -= 1 hint = "too high" if guess > number else "too low" if tries > 0: update_output(f"❌ Your guess is {hint}. {tries} attempts left.") else: update_output(f"💀 Game Over. The number was {number}.") def restart_game(event=None): global number, tries number = random.randint(1, 10) tries = 3 update_output("🔁 Game restarted. Try again!") document.getElementById("submit").addEventListener("click", handle_guess) document.getElementById("restart").addEventListener("click", restart_game)
